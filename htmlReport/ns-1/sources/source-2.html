 package org.example;
 
 import java.sql.*;
 
 public class Cgpa {
 
     static float CgpaCalculate(String DB_URL, String USER, String PASS, String user_id) {
 
         try {
 
             Connection c = null;
             Statement stmt = null;
 
             c = DriverManager.getConnection(DB_URL, USER, PASS);
             stmt = c.createStatement();
             String grade_table = "SELECT * FROM enrollment_list where student_id =" + "\'" + user_id + "\'"
                     + "and course_status = " + "\'" + "completed" + "\'";
             ResultSet rs = stmt.executeQuery(grade_table);
             ResultSetMetaData rsmd = rs.getMetaData();
 
             int columnsNumber = rsmd.getColumnCount();
             // System.out.println(rs.getString(0));
             int total_credits = 0;
             int current_cred = 0;
             int cur_sum = 0;
             while (rs.next()) {
                 // Print one row
                 for (int i = 1; i <= columnsNumber; i++) {
 
                     // System.out.print(rs.getString(i) + " "); // Print one element of a row
 
                     if (i == 4) {
                         current_cred += rs.getInt(i);
                         total_credits += current_cred;
                     }
 
                     if (i == 7) {
                         String grade = rs.getString(i);
                         if (grade.equals("A")) {
                             cur_sum += 10 * current_cred;
                         } else if (grade.equals("A-")) {
                             cur_sum += 9 * current_cred;
                         } else if (grade.equals("B")) {
                             cur_sum += 8 * current_cred;
                         } else if (grade.equals("B-")) {
                             cur_sum += 7 * current_cred;
                         } else if (grade.equals("C")) {
                             cur_sum += 6 * current_cred;
                         } else if (grade.equals("C-")) {
                             cur_sum += 5 * current_cred;
                         } else if (grade.equals("D")) {
                             cur_sum += 4 * current_cred;
                         }
                     }
 
                 }
 
                 // System.out.println();// Move to the next line to print the next row.
 
             }
             rs.close();
 
             // System.out.print("Your current CGPA is : ");
             double ans = Math.round(cur_sum * 100 / total_credits) / 100.00;
             Double D = Double.valueOf(ans);
             float myFloat = D.floatValue();
             return myFloat;
 
         } catch (SQLException e) {
             System.err.println(e.getMessage());
             float ff = 0;
             return ff;
         }
 
     }
 }
